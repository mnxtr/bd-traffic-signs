% ABSTRACT
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\headerbox{Abstract}{name=abstract,column=0,row=0,span=3}{
This capstone project presents a comprehensive deep learning solution for detecting and classifying Bangladeshi traffic signs using \textbf{YOLOv11}. We trained on the Bangladeshi Road Sign Detection Dataset (BRSDD) containing \textbf{8,953 images} across \textbf{29 sign categories}. Our YOLOv11 nano model achieved exceptional performance with \textbf{99.45\% mAP@50}, demonstrating near-perfect detection capability. The project includes complete deployment solutions: a real-time \textbf{Android mobile application} and a \textbf{web-based demo interface}.

\textbf{Keywords:} Traffic Sign Detection, YOLOv11, Object Detection, Deep Learning, Computer Vision, Bangladesh
}

  \item Develop mobile and web deployment solutions
\end{enumerate}

\headerbox{3. Methodology}{name=method,column=1,below=abstract,span=2}{
\textbf{YOLOv11 Architecture:}

\begin{center}
\includegraphics[width=0.9\linewidth]{yolo_architecture.png}
\end{center}

\textbf{Why YOLOv11?}
\begin{itemize}[leftmargin=*,noitemsep,topsep=0pt]
  \item \textbf{State-of-the-art} real-time object detection
  \item \textbf{Single-stage detector} (faster than two-stage methods)
  \item \textbf{Mobile-friendly} variants available
  \item \textbf{Strong transfer learning} from COCO dataset
\end{itemize}

\vspace{0.5em}
\textbf{Training Configuration:}
\begin{multicols}{2}
\begin{itemize}[leftmargin=*,noitemsep,topsep=0pt]
  \item Model: YOLOv11n (Nano)
  \item Epochs: 50
  \item Batch Size: 8
  \item Image Size: 640×640
  \item Device: CPU (AMD Ryzen 7)
  \item Optimizer: AdamW
  \item Learning Rate: 0.01 $\rightarrow$ 0.001
  \item Momentum: 0.937
  \item Weight Decay: 0.0005
\end{itemize}
\end{multicols}

\vspace{0.5em}
\textbf{Loss Functions:}
\begin{itemize}[leftmargin=*,noitemsep,topsep=0pt]
  \item Box Loss: CIoU (Complete Intersection over Union)
  \item Class Loss: Binary Cross-Entropy
  \item DFL Loss: Distribution Focal Loss
\end{itemize}

\vspace{0.5em}
\textbf{Data Augmentation:}
Mosaic (4-image mix), Random flip, Random scale \& translate, HSV color jitter, Random rotation (±10°)
}
